# Array

By completing this chapter, you should be able to answer the following questions:

  1. What is the difference between array and dynamic array?
  2. What is the corresponding built-in data structure of array and dynamic array in your frequently-used language?
  3. How to perform basic operations (initialization, data access, modification, iteration, sort, etc) in an array?
  4. How to perform basic operations (initialization, data access, modification, iteration, sort, addition, deletion, etc) in a dynamic array?
#include <iostream>

int main() {
    // 1. Initialize
    int a0[5];
    int a1[5] = {1, 2, 3};  // other element will be set as the default value
    // 2. Get Length
    int size = sizeof(a1) / sizeof(*a1);
    cout << "The size of a1 is: " << size << endl;
    // 3. Access Element
    cout << "The first element is: " << a1[0] << endl;
    // 4. Iterate all Elements
    cout << "[Version 1] The contents of a1 are:";
    for (int i = 0; i < size; ++i) {
        cout << " " << a1[i];
    }
    cout << endl;
    cout << "[Version 2] The contents of a1 are:";
    for (int& item: a1) {
        cout << " " << item;
    }
    cout << endl;
    // 5. Modify Element
    a1[0] = 4;
    // 6. Sort
    sort(a1, a1 + size);
}

# Dynamic Array

其实是vector拉

#include <iostream>

int main() {
    // 1. initialize
    vector<int> v0;
    vector<int> v1(5, 0);
    // 2. make a copy
    vector<int> v2(v1.begin(), v1.end());
    vector<int> v3(v2);
    // 2. cast an array to a vector
    int a[5] = {0, 1, 2, 3, 4};
    vector<int> v4(a, *(&a + 1));
    // 3. get length
    cout << "The size of v4 is: " << v4.size() << endl;
    // 4. access element
    cout << "The first element in v4 is: " << v4[0] << endl;
    // 5. iterate the vector
    cout << "[Version 1] The contents of v4 are:";
    for (int i = 0; i < v4.size(); ++i) {
        cout << " " << v4[i];
    }
    cout << endl;
    cout << "[Version 2] The contents of v4 are:";
    for (int& item : v4) {
        cout << " " << item;
    }
    cout << endl;
    cout << "[Version 3] The contents of v4 are:";
    for (auto item = v4.begin(); item != v4.end(); ++item) {
        cout << " " << *item;
    }
    cout << endl;
    // 6. modify element
    v4[0] = 5;
    // 7. sort
    sort(v4.begin(), v4.end());
    // 8. add new element at the end of the vector
    v4.push_back(-1);
    // 9. delete the last element
    v4.pop_back();
}

#include <iostream>

template <size_t n, size_t m>
void printArray(int (&a)[n][m]) {
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cout << a[i][j] << " ";
        }
        cout << endl;

}

class Solution {
public:
    void rotate(vector<int>& nums, int k) {
         reverse(nums.begin(),nums.end());                   // reverse the array
        reverse(nums.begin(),nums.begin()+k%nums.size());  // reverse the first k ele
        reverse(nums.begin()+k%nums.size(),nums.end());   // reverse the last n-k ele
    }
};

# Pascal's Triangle I

Input: numRows = 5
Output: [[1],[1,1],[1,2,1],[1,3,3,1],[1,4,6,4,1]]
1
2
Input: numRows = 1
Output: [[1]]

  • 1 <= numRows <= 30

Solution

class Solution {
public:
    vector<vector<int>> generate(int numRows) {
      
        vector<vector<int>> a(numRows);
      for(int i=0;i<numRows;i++)
      {
          a[i].resize(i+1);
          a[i][0] = a[i][i] = 1;
          for(int j=1;j<i;j++)
          {
              a[i][j]=a[i-1][j-1]+a[i-1][j];
          }
      }
        return a;
    }
};

# Remove Duplicates from Sorted Array

Input: nums = [1,1,2]
Output: 2, nums = [1,2]
1
2
Input: nums = [0,0,1,1,1,2,2,3,3,4]
Output: 5, nums = [0,1,2,3,4]
1
2

Constraints:

  • 0 <= nums.length <= 3 * 104
  • -104 <= nums[i] <= 104
  • nums is sorted in ascending order.

Tip : simply change

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if(nums.empty()) // empty nums
            return 0;
        int index =0;
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]!=nums[index])
                nums[++index] = nums[i]; // changed index when different nums
        }
        return index+1;
    }
};
