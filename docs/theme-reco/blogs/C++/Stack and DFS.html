# Target Sum

You are given a list of non-negative integers, a1, a2, ..., an, and a target, S. Now you have 2 symbols + and -. For each integer, you should choose one from + and - as its new symbol.

Input: nums is [1, 1, 1, 1, 1], S is 3. 
Output: 5
Explanation: 

-1+1+1+1+1 = 3
+1-1+1+1+1 = 3
+1+1-1+1+1 = 3
+1+1+1-1+1 = 3
+1+1+1+1-1 = 3

There are 5 ways to assign symbols to make the sum of nums be target 3.

# Solution

class Solution {
public:
    int dp[21][2005];
    int solve(vector<int>&nums,int curr,int idx,int sum)
    {
        if(idx == nums.size()) return (curr==sum)?1:0;
        if(dp[idx][curr+1000]!=-1) return dp[idx][curr+1000];
        return dp[idx][curr+1000] = solve(nums,curr-nums[idx],idx+1,sum)+solve(nums,curr+nums[idx],idx+1,sum);
    }
    int findTargetSumWays(vector<int>& nums, int sum) {
        memset(dp,-1,sizeof(dp));
        return solve(nums,0,0,sum);
    }
};

class Solution {
private:
    string helper(string s, int& pos) {
        string word = ""; int num = 0;
        
        for (; pos < s.size(); pos++) {
            if (s[pos] == '[') {
                string ss = helper(s, ++pos);
                
                for (int i = 0; i < num; i++) word += ss; 
                
                num = 0;
            } else if (s[pos] == ']') {
                return word;
            } else if (isdigit(s[pos])) {
                num = num * 10 + s[pos] - '0';
            } else {
                word += s[pos];
            }
        }
        
        return word;
    }
public:
    string decodeString(string s) {
        int pos = 0; 
        
        return helper(s, pos);
    }
};

# Flood Fill

An image is represented by a 2-D array of integers, each integer representing the pixel value of the image (from 0 to 65535).

Given a coordinate (sr, sc) representing the starting pixel (row and column) of the flood fill, and a pixel value newColor, "flood fill" the image.

To perform a "flood fill", consider the starting pixel, plus any pixels connected 4-directionally to the starting pixel of the same color as the starting pixel, plus any pixels connected 4-directionally to those pixels (also with the same color as the starting pixel), and so on. Replace the color of all of the aforementioned pixels with the newColor.

At the end, return the modified image.

Input: 
image = [[1,1,1],[1,1,0],[1,0,1]]
sr = 1, sc = 1, newColor = 2
Output: [[2,2,2],[2,2,0],[2,0,1]]
Explanation: 
From the center of the image (with position (sr, sc) = (1, 1)), all pixels connected 
by a path of the same color as the starting pixel are colored with the new color.
Note the bottom corner is not colored 2, because it is not 4-directionally connected
to the starting pixel.

class Solution {
public:
    void paint(vector<vector<int>>&image,int sr,int sc, int oldcolor,int newColor,vector<vector<int>>& canvas){
        
        if(sr<0||sc<0||sr>=image.size()||sc>=image[0].size())                          //out of bound condition
            return;
        if(canvas[sr][sc]!=oldcolor)                                                   //color not matches
            return;
        if(newColor==oldcolor)                                                         //if color is same
            return;
        
        canvas[sr][sc]=newColor;
        paint(image,sr-1,sc,image[sr][sc],newColor,canvas);         //check upward
        paint(image,sr,sc+1,image[sr][sc],newColor,canvas);         //check right
        paint(image,sr+1,sc,image[sr][sc],newColor,canvas);         //check downward
        paint(image,sr,sc-1,image[sr][sc],newColor,canvas);         //check left
        
        return;
        
    }
    
    
    vector<vector<int>> floodFill(vector<vector<int>>& image, int sr, int sc, int newColor) {
        
        vector<vector<int>> canvas =image;
        paint(image,sr,sc,image[sr][sc],newColor,canvas);
        return canvas;
        
    }
};

class Solution {
public:
   const int MAX = 10000;
    vector<vector<int>> updateMatrix(vector<vector<int>>& matrix) {
        for(int i = 0; i < matrix.size(); i++)
            for(int j = 0; j < matrix[0].size(); j++)
                if(matrix[i][j])
                    matrix[i][j] = MAX;
        
        for(int i = 0; i < matrix.size(); i++)
            for(int j = 0; j < matrix[0].size(); j++)
                if(matrix[i][j])
                    dfs(matrix, i, j);
        return matrix;
    }
    
    int dfs(vector<vector<int>>& matrix, int r, int c)
    {
        if(r < 0 || c < 0 || r >= matrix.size() || c >= matrix[0].size() || matrix[r][c] == -1)
            return MAX;
        
        if(matrix[r][c] == 0 || matrix[r][c] == 1)
            return matrix[r][c];
        
        int temp = matrix[r][c];
        
        matrix[r][c] = -1;
        
        int left = dfs(matrix, r, c-1);
        int right = dfs(matrix, r, c+1);
        int down = dfs(matrix, r-1, c);
        int up = dfs(matrix, r+1, c);
        
        matrix[r][c] = min({temp, left + 1, right + 1, down + 1, up + 1});
        return matrix[r][c];
    }
};

# Keys and Rooms

Input: [[1],[2],[3],[]]
Output: true
Explanation:  
We start in room 0, and pick up key 1.
We then go to room 1, and pick up key 2.
We then go to room 2, and pick up key 3.
We then go to room 3.  Since we were able to go to every room, we return true.

# Solution

class Solution {
public:
   bool canVisitAllRooms(vector<vector<int>>& rooms) {
        unordered_map<int, bool> room; //Check if this room is open
        bool check = true; //Check if have any new room open
        room[0] = true; //First room always open
        
        while(check){
            check = false; //No new room at first
            for(int i = 0; i < rooms.size(); i++){
                if(!rooms[i].size()){ //If this is empty, skip
                    room[i] = true;
                    continue;
                }
                if(room[i]){ //If this room is open
                    for(int j:rooms[i]){ //Keys
                        if(!room[j]){ //If they relative room is close,
                            room[j] = true; //Open the room
                            check = true; //New room found
                        }
                    }
                }
            }
        }
        
        for(auto i:room){
            if(!i.second){ //If there is still having any close room, return false
                return false;
            }
        }
        
        return true;
    }
};
