sicko2021-08-08 00:00:00technginx

2分钟简单学习nginx配置。

# 配置

简单直接跑起来

server {
	# 监听8080端口,简单体验的话，写不写无所谓
    listen 8080;
    # 简单来说一般使用正则匹配,如果只是体验一下，不需要配置.
    server_name yourdomain.com;
    # 配置静态文件,/表示根目录
    location / {
        root            /usr/share/nginx/html;
        index           index.html;
        try_files       $uri $uri/ /index.html;
