# React函数式编程

函数编程范型的一个特点，就是使用不可变的数据结构。

比如数组中，经常使用push添加新元素到数组中，但在React中，最好使用concat方法，不添加元素，而是创建一个新数组

const t = [1, -1, 3]

const t2 = t.concat(5)

console.log(t)  // [1, -1, 3] is printed
console.log(t2) // [1, -1, 3, 5] is printed

// 不光如此，数组还有许多有用的方法，比如map方法
const t = [1, 2, 3]

const m1 = t.map(value => value * 2) // map返回一个新数组
console.log(m1)   // [2, 4, 6] is printed
const m2 = t.map(value => '<li>' + value + '</li>')
console.log(m2)  
// [ '<li>1</li>', '<li>2</li>', '<li>3</li>' ] is printed

// 亦或者解构赋值
const t = [1, 2, 3, 4, 5]

const [first, second, ...rest] = t

console.log(first, second)  // 1, 2 is printed
console.log(rest)          // [3, 4 ,5] is printed rest:es6新特性

// 解构赋值再深挖一点.
props = {
    name : 'aaa',
    age : 35,
}
const Hello = (props) => {
    const { name ,age } = props
    const bornYear = () => new Date().getFullYear() - age
}
return (
    ...
)
// 如果props具有值,这种写法更好一点
const Hello = ({ name, age }) => {
    const bornYear = () => new Date().getFullYear() - age
}

React中状态不可直接修改原则

// 例如在复杂的状态中，点击事件有两个值，单独触发的场景
const App = () => {
    const [left,setLeft] = useState(0);
    const [right,setRight] = useState(0);
}
// 上面是不太好的写法
// 正确思路应该是这样，统一丢给useState状态管理
const App = () => {
    const [click,setClick] = useState({
        left : 0,
        right : 0
    });
}
// 展开语法优雅的使用
const handleLeftClick = () => setClick({...click,left:click.left+1})
