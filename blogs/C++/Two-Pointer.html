Iterate the array from two ends to the middle

One pointer starts from the beginning while the other pointer starts from the end.

often used in a sorted array

Neat code

class Solution {
public:
    void reverseString(vector<char>& s) {
        int n = s.size();
        for(int i =0;i<n/2;i++) // middle
        {
            swap(s[i],s[n-i-1]);
        }
            
    }
};

# Remove Element

given an value,remove all the elements which equal to the value

Input: nums = [3,2,2,3], val = 3
Output: 2, nums = [2,2]

Input: nums = [0,1,2,2,3,0,4,2], val = 2
Output: 5, nums = [0,1,4,0,3]

class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        // remove only delete the value,does not change the size and capacity
        // return last-one point
        // erase method will change size,but not include capacity
        nums.erase(remove(nums.begin(), nums.end(), val), nums.end());  
        // a = remove(nums.begin,...end()) return last element point(empty)
        // erase(a,b) = delete a->b size
		return nums.size();
    }
};

class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        if(nums.size()==1 || nums.size()==0)return;
            
        int i=0,j =0;
        
        while(j<nums.size()){
       
           if(nums[j]==0) j++;
           
           else{
                swap(nums[i],nums[j]);
                i++;j++;
            }
        }
    }
};
