# String

  • Be familiar with the basic operations in a string, especially the unique operations which we don't have in an array
  • Understand the differences between different comparison functions;
  • Determine whether the string is immutable or not in your favorite language and understand the impact of this feature;
  • Be able to solve basic string-related problems.

For instance, if the length of the string is N, the time complexity of both finding operation and substring operation is O(N).

#include <iostream>

int main() {
    string s1 = "Hello World";
    cout << "s1 is \"Hello World\"" << endl;
    string s2 = s1;
    cout << "s2 is initialized by s1" << endl;
    string s3(s1);
    cout << "s3 is initialized by s1" << endl;
    // compare by '=='
    cout << "Compared by '==':" << endl;
    cout << "s1 and \"Hello World\": " << (s1 == "Hello World") << endl;
    cout << "s1 and s2: " << (s1 == s2) << endl;
    cout << "s1 and s3: " << (s1 == s3) << endl;
    // compare by 'compare'
    cout << "Compared by 'compare':" << endl;
    cout << "s1 and \"Hello World\": " << !s1.compare("Hello World") << endl;
    cout << "s1 and s2: " << !s1.compare(s2) << endl;
    cout << "s1 and s3: " << !s1.compare(s3) << endl;
}

# An Interesting example from leetcode

//solution
class Solution {
public:
    string longestCommonPrefix(vector<string>& words) {
       if (empty(words)) return "";
        string output = "";
        sort(begin(words), end(words));
        auto a = words.front(), b = words.back();
        for (auto i = 0; i < size(a); ++i) {
            if (a[i] == b[i]) { output += a[i]; } else { break; } 
        }
        return output;
    }
};

class Solution {
public:
    string reverseWords(string s) {
       stringstream ss(s);
        string word;
        vector<string> v;
        while(ss>>word)
            v.push_back(word);
        
        string ans = "";
        for(int i=v.size()-1;i>=0;i--)
        {
            ans+=v[i] + ' ';
        }
        ans.erase(ans.end()-1);
        return ans;
    }
};

class Solution {
public:
    string reverseWords(string s) {
        stringstream ss(s);
        string word;
        string ans = "";
        while(ss>>word)
        {
           reverse(word.begin(),word.end());
            ans +=" "+word;
        }
        return ans.substr(1,ans.size()-1);
    }
};
