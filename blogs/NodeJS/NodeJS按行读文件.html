# NodeJS 按行读文件

show the code

import fs from 'fs'
import readline from 'readline'

const fileStream = fs.createReadStream('xx.txt')
const rl = readline.createInterface({
	input: fileStream,
	crlfDelay: Infinity,
})
rl.on('line', (line) => {
	console.log(line)
})
